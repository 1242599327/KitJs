选择器

基本选择器

  • 按照dom id选取

    var el = $kit.el8id('id');//返回id为'id'的元素

  • 按照dom className选取

    var el = $kit.el8cls('class');//返回class为'class'的第一个元素
var elAry = $kit.els8cls('class');//返回class为'class'的元素集合

  • 按照dom name选取

    var el = $kit.el8name('name');//返回name为'name'的第一个元素
var elAry = $kit.els8name('name');//返回name为'name'的元素集合

综合选择器

  • 支持基本的css选择器

    var el = $kit.el8id('id') = $kit.el('#id');//返回id为'id'的元素
var elAry = $kit.els8class('class') = $kit.el('.class');//返回class为'class'的元素集合
var elAry = $kit.el8name('name') = $kit.el('@name');//返回name为'name'的元素集合
